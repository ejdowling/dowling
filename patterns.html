attribute vec2 position; //the position of the point
        varying vec2 vPosition;
        void main(void) { //pre-built function
            vPosition = position;
            gl_Position = vec4(position, 0., 1.); //0. is the z, and 1 is w
        }
    
        precision lowp float;

        varying vec2 vPosition;
        uniform sampler2D sampler2d;
        uniform float idx;

        void main() {
            float pi = 3.14159;
            float x = vPosition.x*0.02;
            float y = vPosition.y*0.02;
            float n = 500.0;
            float m = 900.0;
            float a = cos(n * pi * x) * cos(m * pi * y) - cos(m * pi * x) * cos(n * pi * y);
            vec3 c = mix(vec3(0.1, 0.4, 0.2), vec3(0.8, 0.9, 0.5), floor(a+0.5));
            gl_FragColor = vec4(c, 1);
        }
